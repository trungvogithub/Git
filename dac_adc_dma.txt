void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc){
	A1_read=adc_value[0]/4095.0*1.5;
	f1_read=adc_value[1]*999/4095+1;
	if (((f1_read-f1)>10)|((f1_read-f1)<(-10))) f1=f1_read;
	if (((A1_read-A1)>0.05)|((A1_read-A1)<(-0.05))) A1=A1_read;
	f1_temp=100000/(f1*20)-1;
	TIM6->ARR=(uint16_t)f1_temp;
	for(int i=0;i<=19;i++){
		sinwave[i]=(A1*sin(Dac_step[i]*pi/10)+A1)/3*4095;
	}

}
